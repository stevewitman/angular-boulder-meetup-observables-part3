this.exampleForm.valueChanges
  .pipe(
    switchMap((formValue) =>
      this.http.put('https://jsonplaceholder.typicode.com/posts/1', {
        ...
        body: JSON.stringify({
          id: 1,
          title: formValue,
        }),
        ...
      })
    ),
    takeUntil(this.unsubscribe$)
  )
  .subscribe();
