from([57, 82, 77, 82, 98])
  .pipe(first(num => num >= 70))
  .subscribe(console.log);

from([57, 82, 77, 82, 98])
  .pipe(first(num => num < 50 ))
  .subscribe(console.log);

interval(1000)
  .pipe(first((num) => num > 2))
  .subscribe(console.log);

from([57, 82, 77, 82, 98])
  .pipe(last(num => num < 80))
  .subscribe(console.log);

from([57, 82, 77, 82, 98])
  .pipe(last(num => num === 100))
  .subscribe(console.log);

interval(1000)
  .pipe(
    take(5),
    last(num => num < 3)
  )
  .subscribe(console.log);
