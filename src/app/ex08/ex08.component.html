fromEvent<MouseEvent>(document, 'click')
  .pipe(
    skip(1), // skips the click that starts this example
    map(({ clientX, clientY }) =>
      ({ x: clientX, y: clientY })),
    withLatestFrom(interval(100), interval(3000)),
  )
  .subscribe(([coords, interval1, interval2]) =>
    console.log(
      coords,
      '  t1:', interval1,
      '  t2:', interval2
    )
  );

Click out here and watch the console.

Note: values will only be emitted after, first a click, and also both interval observables have emitted a value (after 3 seconds)
