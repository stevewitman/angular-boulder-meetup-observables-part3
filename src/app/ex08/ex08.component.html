const timer$ = interval(1000)
  .pipe( skip(3) );

const clicker$ = fromEvent(document, 'click')
  .pipe(
    skip(1),
    pluck('clientX')
  );

const sub = combineLatest(timer$, clicker$)
  .subscribe(
    ([timerVal, clickerVal]) =>
    console.log(`Timer: ${timerVal}, Clicker: ${clickerVal}`)
  );
