this.todoForm.valueChanges
  .pipe(
    filter(() => this.todoForm.valid),
    concatMap((changes) => this.saveTodo(changes.title))
  )
  .subscribe();


saveTodo(title: string): Observable<any> {
  return from(fetch(`https://jsonplaceholder.typicode.com/todos/5?_delay=2000`, {
      method: 'PUT',
      body: JSON.stringify({ title: title }),
      headers: {'content-type': 'application/json'},
    })
  );
}
