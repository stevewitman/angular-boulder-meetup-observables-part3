const obsA$ = interval(500)
  .pipe( map(val => 'A-' + val), take(4) );
const obsB$ = interval(1000)
  .pipe( map(val => 'B-' + val), take(3) );
const obsC$ = interval(3000)
  .pipe( map(val => 'C-' + val), take(2) );

forkJoin(obsB$, obsC$, obsA$)
  .subscribe(console.log);
