  )

  .pipe(
    map((n) => n + 1), // convert from zero-based to one-based
    take(4) // emit 4 values then complete
  )
  .subscribe(v => console.log(v));

const timer$: Observable = timer(4500);

interval(1000)
  .pipe(
    map((n) => n + 1),
    takeUntil(timer$) // subscribes to timer$ observable
  )
  .subscribe(v => console.log(v));
